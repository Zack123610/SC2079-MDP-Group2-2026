#ifndef PID_H
#define PID_H

#include <stdint.h>

typedef struct
{
    float kp;
    float ki;

    float integrator;
    float out_min;
    float out_max;
} PIController;

void PI_Init(PIController *pid, float kp, float ki, float out_min, float out_max);

float PI_Update(PIController *pid, float target, float measurement, float dt);

void PI_Reset(PIController *pid);

#endif
