#include "pid.h"

void PI_Init(PIController *pid, float kp, float ki, float out_min, float out_max)
{
    pid->kp = kp;
    pid->ki = ki;
    pid->integrator = 0.0f;
    pid->out_min = out_min;
    pid->out_max = out_max;
}

void PI_Reset(PIController *pid)
{
    pid->integrator = 0.0f;
}

float PI_Update(PIController *pid, float target, float measurement, float dt)
{
    float error = target - measurement;

    /* Integrator */
    pid->integrator += error * dt;

    /* PI output */
    float out = pid->kp * error + pid->ki * pid->integrator;

    /* Clamp output */
    if (out > pid->out_max)
        out = pid->out_max;
    else if (out < pid->out_min)
        out = pid->out_min;

    return out;
}
