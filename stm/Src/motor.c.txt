#include "motor.h"
#include "main.h"

extern TIM_HandleTypeDef htim4;   // Motor A: TIM4 CH3/CH4 (PB8/PB9)
//extern TIM_HandleTypeDef htim9;   // Motor B: TIM9 CH1/CH2 (PE5/PE6)

/* Motor D driver inputs are PE13 / PE14 -> use TIM1 CH3/CH4 (CubeMX must configure this) */
extern TIM_HandleTypeDef htim1;   // <-- MUST exist after CubeMX config

// -------- Motor A (PB8 / PB9) --------
void MotorA_Init(void)
{
    HAL_TIM_PWM_Start(&htim4, TIM_CHANNEL_3);
    HAL_TIM_PWM_Start(&htim4, TIM_CHANNEL_4);
}

void MotorA_Stop(void)
{
    __HAL_TIM_SetCompare(&htim4, TIM_CHANNEL_3, 0);
    __HAL_TIM_SetCompare(&htim4, TIM_CHANNEL_4, 0);
}

void MotorA_Reverse(uint16_t pwm)
{
    __HAL_TIM_SetCompare(&htim4, TIM_CHANNEL_3, pwm);
    __HAL_TIM_SetCompare(&htim4, TIM_CHANNEL_4, 0);
}

void MotorA_Forward(uint16_t pwm)
{
    __HAL_TIM_SetCompare(&htim4, TIM_CHANNEL_3, 0);
    __HAL_TIM_SetCompare(&htim4, TIM_CHANNEL_4, pwm);
}

// -------- Motor D (PE13 / PE14) --------
// We will map:
//  - PE13 = TIM1_CH3
//  - PE14 = TIM1_CH4
void MotorD_Init(void)
{
    HAL_TIM_PWM_Start(&htim1, TIM_CHANNEL_3);
    HAL_TIM_PWM_Start(&htim1, TIM_CHANNEL_4);

    /* TIM1 is advanced timer: ensure outputs are enabled */
    __HAL_TIM_MOE_ENABLE(&htim1);
}


void MotorD_Stop(void)
{
    __HAL_TIM_SetCompare(&htim1, TIM_CHANNEL_3, 0);
    __HAL_TIM_SetCompare(&htim1, TIM_CHANNEL_4, 0);
}

/* SWAPPED: Forward/Reverse mapping */
void MotorD_Forward(uint16_t pwm)
{
    __HAL_TIM_SetCompare(&htim1, TIM_CHANNEL_3, pwm);
    __HAL_TIM_SetCompare(&htim1, TIM_CHANNEL_4, 0);
}

void MotorD_Reverse(uint16_t pwm)
{
    __HAL_TIM_SetCompare(&htim1, TIM_CHANNEL_3, 0);
    __HAL_TIM_SetCompare(&htim1, TIM_CHANNEL_4, pwm);
}
